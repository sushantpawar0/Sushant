Q2. Write a C program to find maximum number in array containing 
five elements of integer type.

#include <stdio.h>

int main() {
    int arr[5]; // Declare an array of 5 integers

    // Read 5 elements from the user
    printf("Enter 5 integer elements:\n");
    for (int i = 0; i < 5; i++) {
        printf("Enter element %d: ", i + 1);
        scanf("%d", &arr[i]);
    }

    // Initialize max with the first element
    int max = arr[0];

    // Find the maximum element in the array
    for (int i = 1; i < 5; i++) {
        if (arr[i] > max) {
            max = arr[i];
        }
    }

    // Display the maximum element
    printf("The maximum number in the array is: %d\n", max);

    return 0;
}
